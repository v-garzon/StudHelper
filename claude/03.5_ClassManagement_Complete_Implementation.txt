@router.post("/", response_model=ClassResponse, status_code=status.HTTP_201_CREATED)
async def create_class(
    class_data: ClassCreate,
    current_user: UserResponse = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Create a new class with settings and permissions"""
    try:
        from app.models import Class, ClassMembership
        
        # Generate unique class code
        class_code = generate_class_code()
        while db.query(Class).filter(Class.class_code == class_code).first():
            class_code = generate_class_code()
        
        # Prepare default permissions (use provided or default to student)
        default_perms = class_data.default_permissions.model_dump() if class_data.default_permissions else get_default_permissions()
        
        # Create class
        new_class = Class(
            name=class_data.name,
            description=class_data.description,
            class_code=class_code,
            owner_id=current_user.id,
            ai_model=class_data.ai_model,
            default_permissions=default_perms,
            sponsorship_enabled=class_data.sponsorship_enabled,
            default_token_limit=class_data.default_token_limit,
            default_token_refresh_rate=class_data.default_token_refresh_rate.value if class_data.default_token_refresh_rate else None
        )
        db.add(new_class)
        db.flush()
        
        # Add owner as manager with full permissions
        manager_perms = get_manager_permissions()
        owner_membership = ClassMembership(
            user_id=current_user.id,
            class_id=new_class.id,
            **manager_perms
        )
        db.add(owner_membership)
        db.commit()
        db.refresh(new_class)
        
        # Add member count
        result = ClassResponse.model_validate(new_class)
        result.member_count = 1
        
        logger.info(f"Class created: {new_class.name} (ID: {new_class.id}) by user {current_user.id}")
        return result
        
    except Exception as e:
        logger.error(f"Error creating class: {e}")
        db.rollback()
        raise HTTPException(status_code=500, detail=f"Internal server error: {str(e)}")

@router.get("/", response_model=List[ClassResponse])
async def get_user_classes(
    current_user: UserResponse = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Get classes where user is owner or member"""
    try:
        from app.models import Class, ClassMembership
        
        # Get classes where user is a member
        memberships = db.query(ClassMembership).filter(
            ClassMembership.user_id == current_user.id
        ).all()
        
        classes = []
        for membership in memberships:
            class_obj = membership.class_obj
            if class_obj.is_active:
                # Count members
                member_count = db.query(ClassMembership).filter(
                    ClassMembership.class_id == class_obj.id
                ).count()
                
                result = ClassResponse.model_validate(class_obj)
                result.member_count = member_count
                classes.append(result)
        
        return classes
        
    except Exception as e:
        logger.error(f"Error getting user classes: {e}")
        raise HTTPException(status_code=500, detail="Internal server error")

@router.post("/join", response_model=ClassResponse)
async def join_class(
    join_data: JoinClassRequest,
    current_user: UserResponse = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Join a class using class code"""
    try:
        from app.models import Class, ClassMembership
        
        # Find class by code
        class_obj = db.query(Class).filter(
            Class.class_code == join_data.class_code.upper(),
            Class.is_active == True
        ).first()
        
        if not class_obj:
            raise HTTPException(status_code=404, detail="Invalid class code")
        
        # Check if already a member
        existing_membership = db.query(ClassMembership).filter(
            ClassMembership.user_id == current_user.id,
            ClassMembership.class_id == class_obj.id
        ).first()
        
        if existing_membership:
            raise HTTPException(status_code=400, detail="Already a member of this class")
        
        # Apply default permissions from class
        default_perms = class_obj.default_permissions if class_obj.default_permissions else get_default_permissions()
        
        # Create membership
        new_membership = ClassMembership(
            user_id=current_user.id,
            class_id=class_obj.id,
            **default_perms
        )
        
        # Apply sponsorship settings if enabled
        if class_obj.sponsorship_enabled:
            new_membership.is_sponsored = True
            new_membership.token_limit = class_obj.default_token_limit
            new_membership.token_refresh_rate = class_obj.default_token_refresh_rate
        
        db.add(new_membership)
        db.commit()
        
        # Count members
        member_count = db.query(ClassMembership).filter(
            ClassMembership.class_id == class_obj.id
        ).count()
        
        result = ClassResponse.model_validate(class_obj)
        result.member_count = member_count
        
        logger.info(f"User {current_user.id} joined class {class_obj.id}")
        return result
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error joining class: {e}")
        db.rollback()
        raise HTTPException(status_code=500, detail="Internal server error")

@router.post("/{class_id}/documents/upload", response_model=List[DocumentUploadResponse])
async def upload_class_documents(
    class_id: int,
    files: List[UploadFile] = File(...),
    current_user: UserResponse = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Upload documents to a class (owner only during creation, or members with can_manage_knowledge)"""
    try:
        from app.models import Class, ClassMembership, Document, DocumentScope
        
        # Check if user has permission
        membership = db.query(ClassMembership).filter(
            ClassMembership.user_id == current_user.id,
            ClassMembership.class_id == class_id
        ).first()
        
        if not membership:
            raise HTTPException(status_code=403, detail="Not a member of this class")
        
        # Check if class exists
        class_obj = db.query(Class).filter(Class.id == class_id).first()
        if not class_obj:
            raise HTTPException(status_code=404, detail="Class not found")
        
        # Check permissions (owner always can, or user with can_manage_knowledge)
        if class_obj.owner_id != current_user.id and not membership.can_manage_knowledge:
            raise HTTPException(status_code=403, detail="No permission to upload class documents")
        
        # Create upload directory
        upload_dir = f"uploads/classes/{class_id}"
        os.makedirs(upload_dir, exist_ok=True)
        
        uploaded_docs = []
        
        for file in files:
            # Save file
            file_path = os.path.join(upload_dir, file.filename)
            
            with open(file_path, "wb") as f:
                content = await file.read()
                f.write(content)
            
            # Create document record
            doc = Document(
                filename=file.filename,
                file_path=file_path,
                file_size=len(content),
                file_type=file.content_type,
                uploaded_by=current_user.id,
                scope=DocumentScope.CLASS,
                class_id=class_id,
                processing_status=ProcessingStatusEnum.PENDING
            )
            db.add(doc)
            db.flush()
            
            uploaded_docs.append(DocumentUploadResponse.model_validate(doc))
            
            logger.info(f"Document uploaded: {file.filename} to class {class_id} by user {current_user.id}")
        
        db.commit()
        
        # TODO: Trigger background processing job for each document
        # For now, documents remain in PENDING status
        
        return uploaded_docs
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error uploading documents: {e}")
        db.rollback()
        raise HTTPException(status_code=500, detail=f"Internal server error: {str(e)}")

@router.get("/{class_id}/chat/availability", response_model=ChatAvailabilityResponse)
async def check_chat_availability(
    class_id: int,
    current_user: UserResponse = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Check if user can create chats in this class"""
    try:
        from app.models import Document, ClassMembership, DocumentScope
        
        # Check if user is member
        membership = db.query(ClassMembership).filter(
            ClassMembership.user_id == current_user.id,
            ClassMembership.class_id == class_id
        ).first()
        
        if not membership:
            raise HTTPException(status_code=403, detail="Not a member of this class")
        
        # Check permission
        if not membership.can_chat:
            return ChatAvailabilityResponse(
                can_chat=False,
                reason="You don't have permission to chat in this class",
                total_count=0
            )
        
        # Get all class documents
        documents = db.query(Document).filter(
            Document.class_id == class_id,
            Document.scope == DocumentScope.CLASS
        ).all()
        
        # If no documents, chat is available
        if len(documents) == 0:
            return ChatAvailabilityResponse(
                can_chat=True,
                total_count=0
            )
        
        # Count by status
        pending_count = sum(1 for d in documents if d.processing_status == ProcessingStatusEnum.PENDING)
        processing_count = sum(1 for d in documents if d.processing_status == ProcessingStatusEnum.PROCESSING)
        completed_count = sum(1 for d in documents if d.processing_status == ProcessingStatusEnum.COMPLETED)
        
        # Chat available only if all documents are completed
        all_completed = completed_count == len(documents)
        
        if not all_completed:
            return ChatAvailabilityResponse(
                can_chat=False,
                reason="Documents are still being processed",
                pending_count=pending_count,
                processing_count=processing_count,
                completed_count=completed_count,
                total_count=len(documents)
            )
        
        return ChatAvailabilityResponse(
            can_chat=True,
            pending_count=0,
            processing_count=0,
            completed_count=completed_count,
            total_count=len(documents)
        )
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error checking chat availability: {e}")
        raise HTTPException(status_code=500, detail="Internal server error")

"""
Document processing service
Handles file uploads, processing, and status tracking
"""
from sqlalchemy.orm import Session
from app.models import Document, ProcessingStatus
from typing import List
import logging

logger = logging.getLogger(__name__)

class DocumentService:
    """Service for document processing operations"""
    
    async def start_processing(self, db: Session, document_id: int):
        """
        Start processing a document
        TODO: Implement actual processing logic
        - Extract text from PDF/DOCX/PPTX
        - Chunk text
        - Generate embeddings
        - Store in vector database
        """
        try:
            doc = db.query(Document).filter(Document.id == document_id).first()
            if not doc:
                logger.error(f"Document {document_id} not found")
                return
            
            doc.processing_status = ProcessingStatus.PROCESSING
            from datetime import datetime
            doc.processing_started_at = datetime.utcnow()
            db.commit()
            
            logger.info(f"Started processing document {document_id}: {doc.filename}")
            
            # TODO: Implement actual processing
            # For now, immediately mark as completed
            doc.processing_status = ProcessingStatus.COMPLETED
            doc.processing_completed_at = datetime.utcnow()
            db.commit()
            
            logger.info(f"Completed processing document {document_id}")
            
        except Exception as e:
            logger.error(f"Error processing document {document_id}: {e}")
            doc = db.query(Document).filter(Document.id == document_id).first()
            if doc:
                doc.processing_status = ProcessingStatus.FAILED
                doc.processing_error = str(e)
                db.commit()
    
    async def process_class_documents(self, db: Session, class_id: int):
        """
        Process all pending documents for a class
        Called after class creation or document upload
        """
        try:
            from app.models import DocumentScope
            
            pending_docs = db.query(Document).filter(
                Document.class_id == class_id,
                Document.scope == DocumentScope.CLASS,
                Document.processing_status == ProcessingStatus.PENDING
            ).all()
            
            logger.info(f"Found {len(pending_docs)} pending documents for class {class_id}")
            
            for doc in pending_docs:
                await self.start_processing(db, doc.id)
                
        except Exception as e:
            logger.error(f"Error processing class documents: {e}")

<script setup>
import { ref, computed, watch } from 'vue'

const props = defineProps({
  modelValue: {
    type: Array,
    default: () => []
  }
})

const emit = defineEmits(['update:modelValue', 'size-exceeded', 'invalid-type'])

const dropZone = ref(null)
const fileInput = ref(null)
const isDragging = ref(false)
const localFiles = ref([...props.modelValue])

const MAX_FILE_SIZE = 50 * 1024 * 1024 // 50MB
const MAX_TOTAL_SIZE = 100 * 1024 * 1024 // 100MB
const ALLOWED_TYPES = [
  'application/pdf',
  'application/msword',
  'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
  'application/vnd.ms-powerpoint',
  'application/vnd.openxmlformats-officedocument.presentationml.presentation',
  'text/plain',
  'text/markdown'
]

const totalSize = computed(() => {
  return localFiles.value.reduce((sum, file) => sum + file.size, 0)
})

watch(() => props.modelValue, (newVal) => {
  localFiles.value = [...newVal]
})

const formatFileSize = (bytes) => {
  if (bytes === 0) return '0 Bytes'
  const k = 1024
  const sizes = ['Bytes', 'KB', 'MB', 'GB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i]
}

const validateFile = (file) => {
  // Check file type
  if (!ALLOWED_TYPES.includes(file.type) && !file.name.match(/\.(pdf|docx?|pptx?|txt|md)$/i)) {
    emit('invalid-type', file.name)
    return false
  }

  // Check individual file size
  if (file.size > MAX_FILE_SIZE) {
    emit('size-exceeded', file.name)
    return false
  }

  // Check total size
  if (totalSize.value + file.size > MAX_TOTAL_SIZE) {
    emit('size-exceeded', 'Total size exceeds 100MB')
    return false
  }

  // Check for duplicates
  if (localFiles.value.some(f => f.name === file.name)) {
    return false
  }

  return true
}

const handleFileSelect = (event) => {
  const files = Array.from(event.target.files)
  addFiles(files)
  event.target.value = '' // Reset input
}

const handleDrop = (event) => {
  isDragging.value = false
  const files = Array.from(event.dataTransfer.files)
  addFiles(files)
}

const addFiles = (files) => {
  const validFiles = files.filter(validateFile)
  localFiles.value = [...localFiles.value, ...validFiles]
  emit('update:modelValue', localFiles.value)
}

const removeFile = (index) => {
  localFiles.value.splice(index, 1)
  emit('update:modelValue', localFiles.value)
}

const triggerFileInput = () => {
  fileInput.value.click()
}
</script>

const validateYouTubeUrl = (url) => {
  const patterns = [
    /(?:youtube\.com\/watch\?v=|youtu\.be\/)([a-zA-Z0-9_-]{11})/,
    /youtube\.com\/embed\/([a-zA-Z0-9_-]{11})/
  ]

  for (let pattern of patterns) {
    const match = url.match(pattern)
    if (match) {
      return { valid: true, videoId: match[1] }
    }
  }

  return { valid: false, error: 'Invalid YouTube URL format' }
}

const addUrl = () => {
  if (!currentUrl.value.trim()) {

    // Name validation
    if (!formData.value.step1.name.trim()) {
      step1Errors.name = 'Class name is required'
    } else if (formData.value.step1.name.trim().length < 3) {
      step1Errors.name = 'Class name must be at least 3 characters'
    } else if (formData.value.step1.name.trim().length > 100) {
      step1Errors.name = 'Class name must be less than 100 characters'
    }

    // File size validation
    const totalSize = formData.value.step1.files.reduce((sum, file) => sum + file.size, 0)
    const maxSize = 100 * 1024 * 1024 // 100MB
    if (totalSize > maxSize) {
      step1Errors.files = 'Total file size exceeds 100MB'
    }

    errors.value.step1 = step1Errors
    return Object.keys(step1Errors).length === 0
  }

  const validateStep2 = () => {
    // Step 2 always valid (UI only for now)
    errors.value.step2 = {}
    return true
  }

  const validateStep3 = () => {
    // Step 3 always valid (UI only for now)
    errors.value.step3 = {}
    return true
  }

  const validateCurrentStep = () => {
    switch (currentStep.value) {
      case 1:
        return validateStep1()
      case 2:
        return validateStep2()
      case 3:
        return validateStep3()
      default:
        return true
    }
  }

  // Navigation
  const nextStep = () => {
    if (!validateCurrentStep()) {
      return
    }
    if (currentStep.value < 3) {
      currentStep.value++
    }
  }

  const previousStep = () => {
    if (currentStep.value > 1) {
      currentStep.value--
    }
  }

  const clearErrors = (step) => {
    if (step) {

    } else {
      errors.value = { step1: {}, step2: {}, step3: {} }
    }
  }

  // Class creation
  const createClass = async () => {
    // Final validation
    if (!validateStep1() || !validateStep2() || !validateStep3()) {
      throw new Error('Validation failed')
    }

    isSubmitting.value = true

    try {
      // Step 1: Create class
      const classData = {
        name: formData.value.step1.name.trim(),
        description: formData.value.step1.description.trim() || null,

        default_token_refresh_rate: formData.value.step3.sponsorshipEnabled ? formData.value.step3.tokenLimits.resetPeriod : null
      }

      // Step 2: Upload files if any
      if (formData.value.step1.files.length > 0) {
        await classService.uploadDocuments(classResponse.id, formData.value.step1.files)
      }

      // TODO: Step 3: Handle YouTube URLs when backend ready
      // Currently YouTube URLs are ignored

      return classResponse
    } catch (error) {
      console.error('Error creating class:', error)
      
      // Set error message
      if (error.response?.data?.detail) {
        errors.value.step1.general = error.response.data.detail
      } else {
        errors.value.step1.general = 'Failed to create class. Please try again.'
      }
      
      throw error
    } finally {
      isSubmitting.value = false
    }
  }

  return {
    currentStep,
    formData,
    errors,
    isSubmitting,
    nextStep,
    previousStep,
    validateCurrentStep,
    clearErrors,
    createClass
  }
}

## Troubleshooting

### Backend Issues

**Migration fails:**
- Check database connection
- Verify you're on the correct revision
- Check for conflicting column names

**File upload fails:**
- Verify `uploads/` directory exists and is writable
- Check file size limits in nginx/server config
- Verify multipart/form-data content type

**Permission errors:**
- Check user is member of class
- Verify permission flags in database
- Check middleware is correctly implemented

### Frontend Issues

**Wizard doesn't open:**
- Check console for errors
- Verify modal is correctly imported
- Check z-index conflicts

**Files won't upload:**
- Check file size (50MB individual, 100MB total)
- Verify file types (PDF, DOCX, PPTX, TXT, MD)
- Check browser console for errors

**Processing view stuck:**
- Check polling is active (every 3 seconds)
- Verify API endpoint returns correct data
- Check document status in database

**Permissions not working:**
- Verify backend returns correct permissions
- Check store is correctly updated
- Verify permission checks in components
